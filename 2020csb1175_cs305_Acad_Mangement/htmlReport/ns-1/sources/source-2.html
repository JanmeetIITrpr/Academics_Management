     static String connectionUser = "postgres";

     public static Connection getConnection () {

             return con;
         }

             Class.forName("org.postgresql.Driver");

             System.out.println("Connected to PSQL database successfully");
         }
         catch(Exception e){
             return null;//not connected
         }
 
         return con;
     }
